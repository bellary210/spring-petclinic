pipeline{
   agent any
     stages {
     stage('clone repo and clean it') {
        steps {
	       sh "git clone https://github.com/bellary210/spring-petclinic.git"
            }
         }
     stage('Build') {
        steps {
               node('master'){
                  sh 'mvn clean package'
                  sh "java -jar target/*.jar"
                  
               }
            }
         }        
      }
}
